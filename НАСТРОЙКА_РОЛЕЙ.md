## Шаг 1 — SQL-скрипт (часть 1: схема, без RLS)

1. В Supabase слева **SQL Editor** → **New query**.
2. Вставьте целиком текст ниже, нажмите **Run** (Ctrl+Enter).

```sql
-- ============================================================
-- РОЛИ И ПРАВА — ЧАСТЬ 1: схема, данные, RPC (БЕЗ включения RLS на app_state)
-- ============================================================

-- 1. Таблицы -------------------------------------------------

create table if not exists public.projects (
  id bigint generated always as identity primary key,
  name text not null,
  app_state_id bigint not null unique,
  created_by uuid not null references auth.users(id),
  created_at timestamptz not null default now()
);
alter table public.projects enable row level security;

create table if not exists public.project_members (
  id bigint generated always as identity primary key,
  project_id bigint not null references public.projects(id) on delete cascade,
  user_id uuid not null references auth.users(id) on delete cascade,
  role text not null check (role in ('admin','member','contractor','client')),
  contractor_name text,
  created_at timestamptz not null default now(),
  unique(project_id, user_id)
);
alter table public.project_members enable row level security;

create table if not exists public.project_invites (
  token uuid primary key default gen_random_uuid(),
  project_id bigint not null references public.projects(id) on delete cascade,
  role text not null check (role in ('admin','member','contractor','client')),
  contractor_name text,
  created_by uuid not null references auth.users(id),
  expires_at timestamptz not null default (now() + interval '30 days'),
  used_at timestamptz
);
alter table public.project_invites enable row level security;

create table if not exists public.project_creation_requests (
  id bigint generated always as identity primary key,
  user_id uuid not null references auth.users(id),
  project_name text not null,
  status text not null default 'pending' check (status in ('pending','approved','rejected')),
  requested_at timestamptz not null default now(),
  decided_at timestamptz
);
alter table public.project_creation_requests enable row level security;

-- 2. Вспомогательные функции для политик (security definer — без рекурсии RLS) ----

create or replace function public.is_project_member(p_project_id bigint)
returns boolean language sql security definer stable as $$
  select exists(select 1 from public.project_members where project_id = p_project_id and user_id = auth.uid());
$$;

create or replace function public.is_project_admin(p_project_id bigint)
returns boolean language sql security definer stable as $$
  select exists(select 1 from public.project_members where project_id = p_project_id and user_id = auth.uid() and role = 'admin');
$$;

create or replace function public.is_system_owner()
returns boolean language sql security definer stable as $$
  select exists(select 1 from auth.users where id = auth.uid() and email = '2150517@mail.ru');
$$;

-- 3. Политики на новые таблицы --------------------------------

drop policy if exists "projects select for members" on public.projects;
create policy "projects select for members" on public.projects for select using (is_project_member(id));

drop policy if exists "members select" on public.project_members;
drop policy if exists "members insert by admin" on public.project_members;
drop policy if exists "members update by admin" on public.project_members;
drop policy if exists "members delete by admin" on public.project_members;
create policy "members select" on public.project_members for select using (is_project_member(project_id));
create policy "members insert by admin" on public.project_members for insert with check (is_project_admin(project_id));
create policy "members update by admin" on public.project_members for update using (is_project_admin(project_id)) with check (is_project_admin(project_id));
create policy "members delete by admin" on public.project_members for delete using (is_project_admin(project_id));

drop policy if exists "invites select" on public.project_invites;
drop policy if exists "invites insert by admin" on public.project_invites;
drop policy if exists "invites delete by admin" on public.project_invites;
-- приглашение открывается по токену (непредсказуемый uuid) — читать может кто угодно,
-- у кого есть точная ссылка, даже без входа (нужно показать «вас пригласили» до логина)
create policy "invites select" on public.project_invites for select using (true);
create policy "invites insert by admin" on public.project_invites for insert with check (is_project_admin(project_id));
create policy "invites delete by admin" on public.project_invites for delete using (is_project_admin(project_id));

drop policy if exists "requests select own or owner" on public.project_creation_requests;
drop policy if exists "requests insert self" on public.project_creation_requests;
create policy "requests select own or owner" on public.project_creation_requests for select using (user_id = auth.uid() or is_system_owner());
create policy "requests insert self" on public.project_creation_requests for insert with check (user_id = auth.uid());

-- 4. RPC-функции -----------------------------------------------

create or replace function public.request_create_project(p_name text)
returns bigint language plpgsql security definer as $$
declare v_id bigint;
begin
  insert into public.project_creation_requests(user_id, project_name)
  values (auth.uid(), p_name)
  returning id into v_id;
  return v_id;
end $$;

create or replace function public.decide_create_project_request(p_request_id bigint, p_approve boolean)
returns bigint language plpgsql security definer as $$
declare
  req record;
  v_app_state_id bigint;
  v_project_id bigint;
begin
  if not public.is_system_owner() then
    raise exception 'Только владелец системы может одобрять заявки';
  end if;
  select * into req from public.project_creation_requests where id = p_request_id and status = 'pending';
  if not found then
    raise exception 'Заявка не найдена или уже обработана';
  end if;
  if p_approve then
    insert into public.app_state(id, data, updated_at) values ((select coalesce(max(id),0)+1 from public.app_state), '{}'::jsonb, now()) returning id into v_app_state_id;
    insert into public.projects(name, app_state_id, created_by) values (req.project_name, v_app_state_id, req.user_id) returning id into v_project_id;
    insert into public.project_members(project_id, user_id, role) values (v_project_id, req.user_id, 'admin');
    update public.project_creation_requests set status = 'approved', decided_at = now() where id = p_request_id;
    return v_app_state_id;
  else
    update public.project_creation_requests set status = 'rejected', decided_at = now() where id = p_request_id;
    return null;
  end if;
end $$;

create or replace function public.accept_invite(p_token uuid)
returns bigint language plpgsql security definer as $$
declare
  inv record;
  v_project_id bigint;
begin
  select * into inv from public.project_invites where token = p_token and used_at is null and expires_at > now();
  if not found then
    raise exception 'Ссылка-приглашение недействительна или уже использована';
  end if;
  insert into public.project_members(project_id, user_id, role, contractor_name)
  values (inv.project_id, auth.uid(), inv.role, inv.contractor_name)
  on conflict (project_id, user_id) do update set role = excluded.role, contractor_name = excluded.contractor_name;
  update public.project_invites set used_at = now() where token = p_token;
  select app_state_id into v_project_id from public.projects where id = inv.project_id;
  return v_project_id; -- id строки app_state, чтобы фронтенд сразу открыл нужный график
end $$;

-- submit_fact: точечная запись факта подрядчиком, без перезаписи всего документа.
-- ПРЕДВАРИТЕЛЬНАЯ версия — буду отдельно проверять и при необходимости чинить на
-- Этапе 5, когда буду подключать реальный ввод факта подрядчиком к этой функции.
create or replace function public.submit_fact(p_app_state_id bigint, p_task_id bigint, p_day_key text, p_value numeric)
returns void language plpgsql security definer as $$
declare
  v_project_id bigint;
  v_role text;
  v_contractor_name text;
  v_task_idx int;
  v_task_contractor text;
begin
  select id into v_project_id from public.projects where app_state_id = p_app_state_id;
  if v_project_id is null then raise exception 'Проект не найден'; end if;

  select role, contractor_name into v_role, v_contractor_name
  from public.project_members where project_id = v_project_id and user_id = auth.uid();
  if v_role is null then raise exception 'Нет доступа к проекту'; end if;

  select (idx - 1), (val->>'contractor') into v_task_idx, v_task_contractor
  from public.app_state, jsonb_array_elements(data->'tasks') with ordinality as t(val, idx)
  where id = p_app_state_id and (val->>'id')::bigint = p_task_id;

  if v_task_idx is null then raise exception 'Работа не найдена'; end if;

  if v_role = 'contractor' and v_task_contractor is distinct from v_contractor_name then
    raise exception 'Можно вносить факт только по своим работам';
  elsif v_role not in ('admin','member','contractor') then
    raise exception 'Недостаточно прав для внесения факта';
  end if;

  update public.app_state
  set data = jsonb_set(
        data, array['tasks', v_task_idx::text, 'factOverrides', 'day', p_day_key],
        to_jsonb(p_value), true),
      updated_at = now()
  where id = p_app_state_id;
end $$;

-- 5. Регистрация текущего графика в таблице projects + вы Администратор.
--    ВАЖНО: формат данных в app_state (id=1) сейчас НЕ трогаем — старый сайт должен
--    продолжать работать без изменений, пока не выложен новый JS (это отдельный шаг 2,
--    его дам, когда буду готов выкладывать новую версию).

insert into public.projects (name, app_state_id, created_by)
select coalesce(data->'projects'->0->>'name', 'Мой график'), 1, (select id from auth.users where email = '2150517@mail.ru')
from public.app_state where id = 1
and not exists (select 1 from public.projects where app_state_id = 1)
returning id;

insert into public.project_members (project_id, user_id, role)
select p.id, (select id from auth.users where email = '2150517@mail.ru'), 'admin'
from public.projects p where p.app_state_id = 1
on conflict (project_id, user_id) do nothing;
```

3. Внизу должно появиться «Success». Если ошибка на шаге регистрации графика (пустое имя
   или что-то не найдено) — пришлите текст ошибки, разберёмся.
4. Проверьте результат — выполните отдельно:
   ```sql
   select * from public.projects;
   select * from public.project_members;
   ```
   В `projects` должна быть одна строка с вашим графиком (`app_state_id = 1`), в
   `project_members` — вы с ролью `admin`. Сам `app_state` (id=1) пока не менялся — сайт
   должен открываться и работать в точности как раньше.

## Шаг 2 — включаем реальную защиту данных (RLS на app_state)

✅ Новый JS выложен на сайт и проверен (вход, роль, загрузка графика — всё работает).
Теперь закрываем открытый доступ к самим данным графика.

Выполните в SQL Editor:

```sql
-- ============================================================
-- РОЛИ И ПРАВА — ЧАСТЬ 2: включаем RLS на app_state (закрываем открытый доступ)
-- ============================================================
alter table public.app_state enable row level security;

drop policy if exists "app_state select for members" on public.app_state;
drop policy if exists "app_state update for members" on public.app_state;

create policy "app_state select for members" on public.app_state
  for select using (
    exists(select 1 from public.projects p where p.app_state_id = app_state.id and is_project_member(p.id))
  );

create policy "app_state update for members" on public.app_state
  for update using (
    exists(
      select 1 from public.projects p
      join public.project_members pm on pm.project_id = p.id
      where p.app_state_id = app_state.id and pm.user_id = auth.uid() and pm.role in ('admin','member')
    )
  );
```

После выполнения — обновите страницу графика (F5) и убедитесь, что всё по-прежнему
открывается и сохраняется (вы — Администратор, для вас ничего не должно измениться).

Примечание: `app_state_history` (резервные копии) пока оставляю открытым как есть — там
пока нет привязки к конкретному графику (`app_state_id`), это лучше сделать позже, когда
графиков станет несколько (Этап 3+). Сейчас реального риска нет — это резервные копии, не
рабочие данные, и график всего один.

## Шаг 3 — заявки на создание графика (Этап 3)

Маленькая правка одной уже созданной функции — теперь она возвращает id новой строки
`app_state`, чтобы фронтенд сразу открывал только что созданный график.

```sql
drop function decide_create_project_request(bigint, boolean);

create function public.decide_create_project_request(p_request_id bigint, p_approve boolean)
returns bigint language plpgsql security definer as $$
declare
  req record;
  v_app_state_id bigint;
  v_project_id bigint;
begin
  if not public.is_system_owner() then
    raise exception 'Только владелец системы может одобрять заявки';
  end if;
  select * into req from public.project_creation_requests where id = p_request_id and status = 'pending';
  if not found then
    raise exception 'Заявка не найдена или уже обработана';
  end if;
  if p_approve then
    insert into public.app_state(id, data, updated_at) values ((select coalesce(max(id),0)+1 from public.app_state), '{}'::jsonb, now()) returning id into v_app_state_id;
    insert into public.projects(name, app_state_id, created_by) values (req.project_name, v_app_state_id, req.user_id) returning id into v_project_id;
    insert into public.project_members(project_id, user_id, role) values (v_project_id, req.user_id, 'admin');
    update public.project_creation_requests set status = 'approved', decided_at = now() where id = p_request_id;
    return v_app_state_id;
  else
    update public.project_creation_requests set status = 'rejected', decided_at = now() where id = p_request_id;
    return null;
  end if;
end $$;
```

## Дополнительно (выполнено по ходу) — таблица app_state изначально была «на одну строку»

При первом создании второго графика обнаружилось: `app_state.id` имел значение по
умолчанию, жёстко равное `1`, и ограничение `single_row`, буквально запрещавшее больше
одной строки (наследие первоначального дизайна «одна строка на всё приложение»). Снято:
```sql
alter table public.app_state drop constraint single_row;
```
(Функция `decide_create_project_request` при вставке новой строки сама вычисляет id как
`max(id)+1`, поэтому автоинкремент/секвенция не нужны.)

## Шаг 4 — участники и приглашения (Этап 4)

`auth.users` (email людей) не читается напрямую с фронтенда (не в публичной схеме) —
поэтому email участника сохраняем прямо в `project_members` в момент присоединения.

Выполните в SQL Editor:

```sql
-- 1. Новая колонка + разовое заполнение для уже существующих участников
alter table public.project_members add column if not exists user_email text;

update public.project_members pm
set user_email = u.email
from auth.users u
where pm.user_id = u.id and pm.user_email is null;

-- 2. accept_invite и decide_create_project_request теперь тоже сохраняют email
create or replace function public.accept_invite(p_token uuid)
returns bigint language plpgsql security definer as $$
declare
  inv record;
  v_app_state_id bigint;
  v_email text;
begin
  select * into inv from public.project_invites where token = p_token and used_at is null and expires_at > now();
  if not found then
    raise exception 'Ссылка-приглашение недействительна или уже использована';
  end if;
  select email into v_email from auth.users where id = auth.uid();
  insert into public.project_members(project_id, user_id, role, contractor_name, user_email)
  values (inv.project_id, auth.uid(), inv.role, inv.contractor_name, v_email)
  on conflict (project_id, user_id) do update set role = excluded.role, contractor_name = excluded.contractor_name, user_email = excluded.user_email;
  update public.project_invites set used_at = now() where token = p_token;
  select app_state_id into v_app_state_id from public.projects where id = inv.project_id;
  return v_app_state_id;
end $$;

create or replace function public.decide_create_project_request(p_request_id bigint, p_approve boolean)
returns bigint language plpgsql security definer as $$
declare
  req record;
  v_app_state_id bigint;
  v_project_id bigint;
  v_email text;
begin
  if not public.is_system_owner() then
    raise exception 'Только владелец системы может одобрять заявки';
  end if;
  select * into req from public.project_creation_requests where id = p_request_id and status = 'pending';
  if not found then
    raise exception 'Заявка не найдена или уже обработана';
  end if;
  if p_approve then
    select email into v_email from auth.users where id = req.user_id;
    insert into public.app_state(id, data, updated_at) values ((select coalesce(max(id),0)+1 from public.app_state), '{}'::jsonb, now()) returning id into v_app_state_id;
    insert into public.projects(name, app_state_id, created_by) values (req.project_name, v_app_state_id, req.user_id) returning id into v_project_id;
    insert into public.project_members(project_id, user_id, role, user_email) values (v_project_id, req.user_id, 'admin', v_email);
    update public.project_creation_requests set status = 'approved', decided_at = now() where id = p_request_id;
    return v_app_state_id;
  else
    update public.project_creation_requests set status = 'rejected', decided_at = now() where id = p_request_id;
    return null;
  end if;
end $$;
```

Проверка:
```sql
select id, user_id, role, user_email from public.project_members;
```
У всех строк должен появиться `user_email` (для существующих участников — заполнится задним числом).
