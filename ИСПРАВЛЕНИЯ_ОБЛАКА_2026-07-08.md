## 1. Починка ввода факта подрядчиком (`submit_fact`)

Старая функция искала работы не по тому пути в документе (`data→tasks` вместо
`data→projects[0]→tasks`) и не пересчитывала общий факт работы. Новая функция
`submit_fact_days` принимает сразу все дни периода одним вызовом, проверяет права
так же, как раньше, обновляет дни, пересчитывает `factVol`/`factPct` и метки времени.

```sql
-- старую нерабочую функцию убираем
drop function if exists public.submit_fact(bigint, bigint, text, numeric);

create or replace function public.submit_fact_days(p_app_state_id bigint, p_task_id bigint, p_days jsonb)
returns void language plpgsql security definer as $$
declare
  v_project_id bigint;
  v_role text;
  v_contractor_name text;
  v_task_idx int;
  v_task jsonb;
  v_day_ov jsonb;
  v_old_sum numeric := 0;
  v_new_sum numeric := 0;
  v_key text;
  v_val jsonb;
  v_volume numeric;
  v_fact_vol numeric;
begin
  if p_days is null or jsonb_typeof(p_days) <> 'object' then
    raise exception 'Неверный формат данных факта';
  end if;

  select id into v_project_id from public.projects where app_state_id = p_app_state_id;
  if v_project_id is null then raise exception 'Проект не найден'; end if;

  select role, contractor_name into v_role, v_contractor_name
  from public.project_members where project_id = v_project_id and user_id = auth.uid();
  if v_role is null then raise exception 'Нет доступа к проекту'; end if;
  if v_role not in ('admin','member','contractor') then
    raise exception 'Недостаточно прав для внесения факта';
  end if;

  -- работы лежат в data -> projects[0] -> tasks (в старой функции путь был неверный)
  select (idx - 1), val into v_task_idx, v_task
  from public.app_state, jsonb_array_elements(data->'projects'->0->'tasks') with ordinality as t(val, idx)
  where id = p_app_state_id and (val->>'id')::bigint = p_task_id;
  if v_task_idx is null then raise exception 'Работа не найдена'; end if;

  if v_role = 'contractor' and (v_task->>'contractor') is distinct from v_contractor_name then
    raise exception 'Можно вносить факт только по своим работам';
  end if;

  -- обновляем дневные значения; null в p_days означает «удалить день»,
  -- общий факт меняем по разнице сумм — так же, как это делает страница
  v_day_ov := coalesce(v_task->'factOverrides'->'day', '{}'::jsonb);
  for v_key, v_val in select key, value from jsonb_each(p_days) loop
    if v_day_ov ? v_key then
      v_old_sum := v_old_sum + coalesce((v_day_ov->>v_key)::numeric, 0);
    end if;
    if jsonb_typeof(v_val) = 'null' then
      v_day_ov := v_day_ov - v_key;
    else
      v_day_ov := jsonb_set(v_day_ov, array[v_key], v_val, true);
      v_new_sum := v_new_sum + (v_val)::text::numeric;
    end if;
  end loop;

  v_volume := coalesce((v_task->>'volume')::numeric, 0);
  v_fact_vol := greatest(0, coalesce((v_task->>'factVol')::numeric, 0) - v_old_sum + v_new_sum);

  v_task := jsonb_set(v_task, '{factOverrides}', coalesce(v_task->'factOverrides', '{}'::jsonb), true);
  v_task := jsonb_set(v_task, '{factOverrides,day}', v_day_ov, true);
  v_task := jsonb_set(v_task, '{factVol}', to_jsonb(v_fact_vol), true);
  v_task := jsonb_set(v_task, '{factPct}',
    to_jsonb(case when v_volume > 0 then round(v_fact_vol / v_volume * 100) else 0 end), true);

  update public.app_state
  set data = jsonb_set(
        jsonb_set(data, array['projects','0','tasks', v_task_idx::text], v_task),
        '{updatedAt}', to_jsonb((extract(epoch from now()) * 1000)::bigint), true),
      updated_at = now()
  where id = p_app_state_id;
end $$;
```

## 2. Резервные копии — привязка к конкретному графику

Раньше история версий была общей на все графики: в списке «Резервные копии» снимки
всех графиков шли вперемешку, и можно было случайно восстановить копию одного графика
поверх другого. Теперь у каждого снимка есть `app_state_id`.

```sql
alter table public.app_state_history
  add column if not exists app_state_id bigint;

create index if not exists app_state_history_app_state_id_idx
  on public.app_state_history (app_state_id, id desc);
```

Старые снимки (сделанные до этого исправления) остаются в таблице, но в списке копий
больше не показываются — у них неизвестно, к какому графику они относятся, и
восстанавливать их вслепую опасно. Если графиком до 7 июля был только один (id = 1),
можно пометить старые снимки как его и вернуть их в список — для этого выполните
дополнительно:

```sql
-- НЕОБЯЗАТЕЛЬНО: только если до 7 июля 2026 существовал единственный график (id = 1)
update public.app_state_history set app_state_id = 1 where app_state_id is null;
```

## 3. Полное удаление графика (`delete_project`)

Кнопка «Удалить график» в настройках раньше не работала (удаляла только из памяти
страницы, облако не трогала). Теперь она вызывает эту функцию: только администратор
графика может удалить его — вместе с участниками, приглашениями, резервными копиями
и самими данными.

```sql
create or replace function public.delete_project(p_project_id bigint)
returns void language plpgsql security definer as $$
declare
  v_role text;
  v_app_state_id bigint;
begin
  select role into v_role from public.project_members
  where project_id = p_project_id and user_id = auth.uid();
  if v_role is distinct from 'admin' then
    raise exception 'Удалить график может только администратор';
  end if;

  select app_state_id into v_app_state_id from public.projects where id = p_project_id;

  delete from public.project_invites where project_id = p_project_id;
  delete from public.project_members where project_id = p_project_id;
  if v_app_state_id is not null then
    delete from public.app_state_history where app_state_id = v_app_state_id;
    delete from public.app_state where id = v_app_state_id;
  end if;
  delete from public.projects where id = p_project_id;
end $$;
```

## 4. Заявки на график — с полным содержимым мастера

Раньше в заявке сохранялось только название графика: выбранная структура разделов,
период и рабочие дни выбрасывались, и после одобрения человек получал пустой график.
Теперь заявка несёт подготовленный график целиком, и при одобрении он сразу
заполняется.

```sql
alter table public.project_creation_requests
  add column if not exists payload jsonb;

-- сигнатура меняется — старую функцию с одним аргументом убираем,
-- чтобы не осталось двух одноимённых
drop function if exists public.request_create_project(text);

create or replace function public.request_create_project(p_name text, p_payload jsonb default null)
returns bigint language plpgsql security definer as $$
declare v_id bigint;
begin
  insert into public.project_creation_requests(user_id, project_name, payload)
  values (auth.uid(), p_name, p_payload)
  returning id into v_id;
  return v_id;
end $$;

create or replace function public.decide_create_project_request(p_request_id bigint, p_approve boolean)
returns bigint language plpgsql security definer as $$
declare
  req record;
  v_app_state_id bigint;
  v_project_id bigint;
begin
  if not public.is_system_owner() then
    raise exception 'Только владелец системы может одобрять заявки';
  end if;
  select * into req from public.project_creation_requests where id = p_request_id and status = 'pending';
  if not found then
    raise exception 'Заявка не найдена или уже обработана';
  end if;
  if p_approve then
    -- новый график сразу получает содержимое, подготовленное в мастере (если оно было в заявке)
    insert into public.app_state(id, data, updated_at)
    values ((select coalesce(max(id),0)+1 from public.app_state), coalesce(req.payload, '{}'::jsonb), now())
    returning id into v_app_state_id;
    insert into public.projects(name, app_state_id, created_by) values (req.project_name, v_app_state_id, req.user_id) returning id into v_project_id;
    insert into public.project_members(project_id, user_id, role) values (v_project_id, req.user_id, 'admin');
    update public.project_creation_requests set status = 'approved', decided_at = now() where id = p_request_id;
    return v_app_state_id;
  else
    update public.project_creation_requests set status = 'rejected', decided_at = now() where id = p_request_id;
    return null;
  end if;
end $$;
```
