## Шаг 1 — SQL-скрипт (схема компаний)

1. В Supabase слева **SQL Editor** → **New query**.
2. Вставьте целиком текст ниже, нажмите **Run** (Ctrl+Enter).

```sql
-- ============================================================
-- ЭТАП 6.1 — КОМПАНИИ: профили, компании, сотрудники, приглашения
-- Только добавление. Существующие таблицы не трогаются.
-- ============================================================

-- 1. Профили пользователей -----------------------------------
-- auth.users не читается с фронтенда напрямую, поэтому имя/почта/публичный
-- номер пользователя денормализуем в public.profiles (как уже сделано с user_email
-- в project_members). public_id — короткий человекочитаемый номер для поиска.

create table if not exists public.profiles (
  user_id    uuid primary key references auth.users(id) on delete cascade,
  public_id  bigint generated always as identity (start with 1001) unique,
  email      text,
  full_name  text,
  created_at timestamptz not null default now()
);
alter table public.profiles enable row level security;

-- профиль создаётся автоматически при регистрации нового пользователя
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer set search_path = public as $$
begin
  insert into public.profiles(user_id, email)
  values (new.id, new.email)
  on conflict (user_id) do nothing;
  return new;
end $$;

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute function public.handle_new_user();

-- разовое заполнение для уже существующих пользователей
insert into public.profiles(user_id, email)
select id, email from auth.users
on conflict (user_id) do nothing;

-- 2. Компании ------------------------------------------------
-- Ключевой классификатор — ИНН (уникальный). owner_user_id — владелец компании
-- в системе (первый, кто её зарегистрировал). Может быть NULL, если компанию как
-- «пустышку» завёл РП, назначая подрядчика (владелец появится, приняв приглашение).

create table if not exists public.companies (
  id            bigint generated always as identity primary key,
  inn           text not null unique,
  name          text not null,
  owner_user_id uuid references auth.users(id),
  created_at    timestamptz not null default now()
);
alter table public.companies enable row level security;

create table if not exists public.company_members (
  id         bigint generated always as identity primary key,
  company_id bigint not null references public.companies(id) on delete cascade,
  user_id    uuid   not null references auth.users(id) on delete cascade,
  user_email text,
  added_at   timestamptz not null default now(),
  unique(company_id, user_id)
);
alter table public.company_members enable row level security;

create table if not exists public.company_invites (
  token      uuid primary key default gen_random_uuid(),
  company_id bigint not null references public.companies(id) on delete cascade,
  created_by uuid not null references auth.users(id),
  expires_at timestamptz not null default (now() + interval '30 days'),
  used_at    timestamptz
);
alter table public.company_invites enable row level security;

-- 3. Точечное назначение подрядчика (таблица под шаг 6.3; RPC подключим позже) ---
-- scope_type: 'section' (раздел, cat:true) или 'task' (отдельная работа).
-- scope_id — id узла внутри массива работ графика (у разделов и работ он стабильный).

create table if not exists public.work_assignments (
  id                  bigint generated always as identity primary key,
  app_state_id        bigint not null,
  scope_type          text not null check (scope_type in ('section','task')),
  scope_id            bigint not null,
  company_id          bigint not null references public.companies(id) on delete cascade,
  responsible_user_id uuid references auth.users(id),
  assigned_by         uuid not null references auth.users(id),
  assigned_at         timestamptz not null default now(),
  unique(app_state_id, scope_type, scope_id, company_id)
);
alter table public.work_assignments enable row level security;

-- 4. Проверка ИНН (контрольная сумма 10 или 12 цифр) ----------
create or replace function public.is_valid_inn(p_inn text)
returns boolean language plpgsql immutable as $$
declare
  d int[];
  i int;
  n10 int; n11 int; n12 int;
  w10  int[] := array[2,4,10,3,5,9,4,6,8];
  w11  int[] := array[7,2,4,10,3,5,9,4,6,8];
  w12  int[] := array[3,7,2,4,10,3,5,9,4,6,8];
  s int;
begin
  if p_inn is null or p_inn !~ '^[0-9]+$' then return false; end if;
  if length(p_inn) not in (10, 12) then return false; end if;
  d := array(select (substring(p_inn from g for 1))::int from generate_series(1, length(p_inn)) g);
  if length(p_inn) = 10 then
    s := 0;
    for i in 1..9 loop s := s + d[i] * w10[i]; end loop;
    n10 := (s % 11) % 10;
    return n10 = d[10];
  else
    s := 0;
    for i in 1..10 loop s := s + d[i] * w11[i]; end loop;
    n11 := (s % 11) % 10;
    s := 0;
    for i in 1..11 loop s := s + d[i] * w12[i]; end loop;
    n12 := (s % 11) % 10;
    return n11 = d[11] and n12 = d[12];
  end if;
end $$;

-- 5. Хелперы для политик (security definer — без рекурсии RLS) --
create or replace function public.is_company_member(p_company_id bigint)
returns boolean language sql security definer stable as $$
  select exists(select 1 from public.company_members
                where company_id = p_company_id and user_id = auth.uid());
$$;

create or replace function public.is_company_owner(p_company_id bigint)
returns boolean language sql security definer stable as $$
  select exists(select 1 from public.companies
                where id = p_company_id and owner_user_id = auth.uid());
$$;

-- 6. Политики ------------------------------------------------

-- profiles: любой вошедший может прочитать профиль (нужно, чтобы РП находил людей
-- по номеру и видел ФИО сотрудников подрядчика); менять — только свой.
drop policy if exists "profiles select all" on public.profiles;
drop policy if exists "profiles update own" on public.profiles;
create policy "profiles select all" on public.profiles for select using (auth.uid() is not null);
create policy "profiles update own" on public.profiles for update using (user_id = auth.uid()) with check (user_id = auth.uid());

-- companies: читать может любой вошедший (поиск по ИНН, показ названия);
-- создание/изменение — через RPC (security definer) либо владельцем.
drop policy if exists "companies select all" on public.companies;
drop policy if exists "companies update owner" on public.companies;
create policy "companies select all" on public.companies for select using (auth.uid() is not null);
create policy "companies update owner" on public.companies for update using (is_company_owner(id)) with check (is_company_owner(id));

-- company_members: читать может любой вошедший (РП выбирает ответственного из
-- сотрудников подрядчика); менять — владелец компании (вставка сотрудника при
-- приёме приглашения идёт через RPC security definer, минуя это ограничение).
drop policy if exists "company_members select all" on public.company_members;
drop policy if exists "company_members delete owner" on public.company_members;
create policy "company_members select all" on public.company_members for select using (auth.uid() is not null);
create policy "company_members delete owner" on public.company_members for delete using (is_company_owner(company_id));

-- company_invites: открывается по токену (непредсказуемый uuid) — как project_invites.
drop policy if exists "company_invites select" on public.company_invites;
drop policy if exists "company_invites insert owner" on public.company_invites;
drop policy if exists "company_invites delete owner" on public.company_invites;
create policy "company_invites select" on public.company_invites for select using (true);
create policy "company_invites insert owner" on public.company_invites for insert with check (is_company_owner(company_id));
create policy "company_invites delete owner" on public.company_invites for delete using (is_company_owner(company_id));

-- work_assignments: видят участники графика; создаёт/меняет админ графика (РП).
drop policy if exists "wa select members" on public.work_assignments;
drop policy if exists "wa write admin" on public.work_assignments;
create policy "wa select members" on public.work_assignments for select using (
  exists(select 1 from public.projects p where p.app_state_id = work_assignments.app_state_id and is_project_member(p.id))
);
create policy "wa write admin" on public.work_assignments for all using (
  exists(select 1 from public.projects p where p.app_state_id = work_assignments.app_state_id and is_project_admin(p.id))
) with check (
  exists(select 1 from public.projects p where p.app_state_id = work_assignments.app_state_id and is_project_admin(p.id))
);

-- создать СВОЮ компанию (я — владелец и первый сотрудник)
create or replace function public.create_my_company(p_inn text, p_name text)
returns bigint language plpgsql security definer as $$
declare v_id bigint; v_email text;
begin
  if not public.is_valid_inn(p_inn) then
    raise exception 'Некорректный ИНН (ожидается 10 или 12 цифр с верной контрольной суммой)';
  end if;
  if trim(coalesce(p_name,'')) = '' then
    raise exception 'Укажите название компании';
  end if;
  select id into v_id from public.companies where inn = p_inn;
  if v_id is not null then
    -- компания уже есть: если без владельца — становлюсь владельцем; иначе просто
    -- присоединяюсь сотрудником
    update public.companies set owner_user_id = coalesce(owner_user_id, auth.uid()) where id = v_id;
  else
    insert into public.companies(inn, name, owner_user_id)
    values (p_inn, trim(p_name), auth.uid()) returning id into v_id;
  end if;
  select email into v_email from auth.users where id = auth.uid();
  insert into public.company_members(company_id, user_id, user_email)
  values (v_id, auth.uid(), v_email)
  on conflict (company_id, user_id) do nothing;
  return v_id;
end $$;

-- мои компании (id, инн, название, владелец ли я)
create or replace function public.get_my_companies()
returns table(id bigint, inn text, name text, is_owner boolean)
language sql security definer stable as $$
  select c.id, c.inn, c.name, (c.owner_user_id = auth.uid())
  from public.companies c
  join public.company_members m on m.company_id = c.id
  where m.user_id = auth.uid()
  order by c.name;
$$;

-- найти компанию по ИНН (для РП, назначающего подрядчика)
create or replace function public.find_company_by_inn(p_inn text)
returns table(id bigint, inn text, name text, has_owner boolean, members_count bigint)
language sql security definer stable as $$
  select c.id, c.inn, c.name, (c.owner_user_id is not null),
         (select count(*) from public.company_members m where m.company_id = c.id)
  from public.companies c where c.inn = p_inn;
$$;

-- сотрудники компании (для показа списка и выбора ответственного)
create or replace function public.get_company_members(p_company_id bigint)
returns table(user_id uuid, email text, full_name text, public_id bigint, is_owner boolean)
language sql security definer stable as $$
  select m.user_id, coalesce(p.email, m.user_email), p.full_name, p.public_id,
         (c.owner_user_id = m.user_id)
  from public.company_members m
  join public.companies c on c.id = m.company_id
  left join public.profiles p on p.user_id = m.user_id
  where m.company_id = p_company_id
  order by (c.owner_user_id = m.user_id) desc, coalesce(p.full_name, p.email, m.user_email);
$$;

-- пригласить сотрудника в компанию (только владелец) -> токен
create or replace function public.invite_to_company(p_company_id bigint)
returns uuid language plpgsql security definer as $$
declare v_token uuid;
begin
  if not public.is_company_owner(p_company_id) then
    raise exception 'Приглашать сотрудников может только владелец компании';
  end if;
  insert into public.company_invites(company_id, created_by)
  values (p_company_id, auth.uid()) returning token into v_token;
  return v_token;
end $$;

-- принять приглашение в компанию. Если у компании нет владельца — принявший
-- становится владельцем (случай «пустышки», заведённой РП).
create or replace function public.accept_company_invite(p_token uuid)
returns bigint language plpgsql security definer as $$
declare inv record; v_email text;
begin
  select * into inv from public.company_invites
  where token = p_token and used_at is null and expires_at > now();
  if not found then
    raise exception 'Ссылка-приглашение в компанию недействительна или уже использована';
  end if;
  select email into v_email from auth.users where id = auth.uid();
  insert into public.company_members(company_id, user_id, user_email)
  values (inv.company_id, auth.uid(), v_email)
  on conflict (company_id, user_id) do update set user_email = excluded.user_email;
  update public.companies set owner_user_id = coalesce(owner_user_id, auth.uid())
  where id = inv.company_id;
  update public.company_invites set used_at = now() where token = p_token;
  return inv.company_id;
end $$;

-- профиль текущего пользователя (создать при отсутствии + обновить почту/имя)
create or replace function public.ensure_profile(p_full_name text default null)
returns void language plpgsql security definer as $$
declare v_email text;
begin
  select email into v_email from auth.users where id = auth.uid();
  insert into public.profiles(user_id, email, full_name)
  values (auth.uid(), v_email, nullif(trim(coalesce(p_full_name,'')),''))
  on conflict (user_id) do update
    set email = excluded.email,
        full_name = coalesce(nullif(trim(coalesce(p_full_name,'')),''), public.profiles.full_name);
end $$;
```

3. Внизу должно появиться «Success. No rows returned».

## Шаг 2 (6.3) — назначение подрядчика на раздел/работу (RPC)

Добавляет функции для интерфейса РП. Существующие таблицы не меняет.

```sql
-- ensure_company: найти компанию по ИНН или создать «пустышку» (без владельца).
-- Используется РП, когда назначает подрядчика, которого ещё нет в системе.
create or replace function public.ensure_company(p_inn text, p_name text)
returns bigint language plpgsql security definer as $$
declare v_id bigint;
begin
  if not public.is_valid_inn(p_inn) then raise exception 'Некорректный ИНН'; end if;
  select id into v_id from public.companies where inn = p_inn;
  if v_id is null then
    if trim(coalesce(p_name,'')) = '' then raise exception 'Для новой компании укажите название'; end if;
    insert into public.companies(inn, name) values (p_inn, trim(p_name)) returning id into v_id;
  end if;
  return v_id;
end $$;

-- invite_to_company: разрешаем приглашать и владельцу, и в «пустышку» без владельца
-- (тогда первый принявший станет владельцем) — чтобы РП мог позвать первого сотрудника.
create or replace function public.invite_to_company(p_company_id bigint)
returns uuid language plpgsql security definer as $$
declare v_token uuid; v_owner uuid;
begin
  select owner_user_id into v_owner from public.companies where id = p_company_id;
  if not (v_owner = auth.uid() or v_owner is null) then
    raise exception 'Приглашать сотрудников может только владелец компании';
  end if;
  insert into public.company_invites(company_id, created_by)
  values (p_company_id, auth.uid()) returning token into v_token;
  return v_token;
end $$;

-- assign_work: РП назначает компанию (+ ответственного) на раздел или работу.
-- Одна компания на узел: прежнее назначение этого узла заменяется.
create or replace function public.assign_work(
  p_app_state_id bigint, p_scope_type text, p_scope_id bigint,
  p_company_id bigint, p_responsible_user_id uuid)
returns bigint language plpgsql security definer as $$
declare v_id bigint;
begin
  if not exists(select 1 from public.projects p
                where p.app_state_id = p_app_state_id and public.is_project_admin(p.id)) then
    raise exception 'Назначать подрядчиков может только руководитель проекта';
  end if;
  if p_scope_type not in ('section','task') then raise exception 'Неверный тип узла'; end if;
  if p_responsible_user_id is not null and not exists(
       select 1 from public.company_members m
       where m.company_id = p_company_id and m.user_id = p_responsible_user_id) then
    raise exception 'Ответственный должен быть сотрудником выбранной компании';
  end if;
  delete from public.work_assignments
    where app_state_id = p_app_state_id and scope_type = p_scope_type and scope_id = p_scope_id;
  insert into public.work_assignments(app_state_id, scope_type, scope_id, company_id, responsible_user_id, assigned_by)
  values (p_app_state_id, p_scope_type, p_scope_id, p_company_id, p_responsible_user_id, auth.uid())
  returning id into v_id;
  return v_id;
end $$;

-- get_work_assignments: назначения графика с названием компании и именем ответственного.
-- Видно только участникам графика (иначе пусто).
create or replace function public.get_work_assignments(p_app_state_id bigint)
returns table(id bigint, scope_type text, scope_id bigint, company_id bigint,
              company_name text, company_inn text, responsible_user_id uuid, responsible_name text)
language sql security definer stable as $$
  select w.id, w.scope_type, w.scope_id, w.company_id, c.name, c.inn,
         w.responsible_user_id, coalesce(p.full_name, p.email)
  from public.work_assignments w
  join public.companies c on c.id = w.company_id
  left join public.profiles p on p.user_id = w.responsible_user_id
  where w.app_state_id = p_app_state_id
    and exists(select 1 from public.projects pr
               where pr.app_state_id = p_app_state_id and public.is_project_member(pr.id));
$$;
```

## Шаг 3 (6.4) — ввод факта подрядчиком по назначенным работам

Обновляет `submit_fact_days`: подрядчик может вносить факт по работе, только если он
назначен по ней **ответственным** в `work_assignments` (раньше проверялось текстовое поле
`contractor`). Остальная логика функции без изменений.

```sql
create or replace function public.submit_fact_days(p_app_state_id bigint, p_task_id bigint, p_days jsonb)
returns void language plpgsql security definer as $$
declare
  v_project_id bigint;
  v_role text;
  v_task_idx int;
  v_task jsonb;
  v_day_ov jsonb;
  v_old_sum numeric := 0;
  v_new_sum numeric := 0;
  v_key text;
  v_val jsonb;
  v_volume numeric;
  v_fact_vol numeric;
begin
  if p_days is null or jsonb_typeof(p_days) <> 'object' then
    raise exception 'Неверный формат данных факта';
  end if;

  select id into v_project_id from public.projects where app_state_id = p_app_state_id;
  if v_project_id is null then raise exception 'Проект не найден'; end if;

  select role into v_role
  from public.project_members where project_id = v_project_id and user_id = auth.uid();
  if v_role is null then raise exception 'Нет доступа к проекту'; end if;
  if v_role not in ('admin','member','contractor') then
    raise exception 'Недостаточно прав для внесения факта';
  end if;

  -- подрядчик — только по работам, где он назначен ответственным (work_assignments)
  if v_role = 'contractor' and not exists(
       select 1 from public.work_assignments w
       where w.app_state_id = p_app_state_id
         and w.scope_type = 'task'
         and w.scope_id = p_task_id
         and w.responsible_user_id = auth.uid()
     ) then
    raise exception 'Можно вносить факт только по назначенным вам работам';
  end if;

  -- работы лежат в data -> projects[0] -> tasks
  select (idx - 1), val into v_task_idx, v_task
  from public.app_state, jsonb_array_elements(data->'projects'->0->'tasks') with ordinality as t(val, idx)
  where id = p_app_state_id and (val->>'id')::bigint = p_task_id;
  if v_task_idx is null then raise exception 'Работа не найдена'; end if;

  -- обновляем дневные значения; null в p_days означает «удалить день»,
  -- общий факт меняем по разнице сумм — так же, как это делает страница
  v_day_ov := coalesce(v_task->'factOverrides'->'day', '{}'::jsonb);
  for v_key, v_val in select key, value from jsonb_each(p_days) loop
    if v_day_ov ? v_key then
      v_old_sum := v_old_sum + coalesce((v_day_ov->>v_key)::numeric, 0);
    end if;
    if jsonb_typeof(v_val) = 'null' then
      v_day_ov := v_day_ov - v_key;
    else
      v_day_ov := jsonb_set(v_day_ov, array[v_key], v_val, true);
      v_new_sum := v_new_sum + (v_val)::text::numeric;
    end if;
  end loop;

  v_volume := coalesce((v_task->>'volume')::numeric, 0);
  v_fact_vol := greatest(0, coalesce((v_task->>'factVol')::numeric, 0) - v_old_sum + v_new_sum);

  v_task := jsonb_set(v_task, '{factOverrides}', coalesce(v_task->'factOverrides', '{}'::jsonb), true);
  v_task := jsonb_set(v_task, '{factOverrides,day}', v_day_ov, true);
  v_task := jsonb_set(v_task, '{factVol}', to_jsonb(v_fact_vol), true);
  v_task := jsonb_set(v_task, '{factPct}',
    to_jsonb(case when v_volume > 0 then round(v_fact_vol / v_volume * 100) else 0 end), true);

  update public.app_state
  set data = jsonb_set(
        jsonb_set(data, array['projects','0','tasks', v_task_idx::text], v_task),
        '{updatedAt}', to_jsonb((extract(epoch from now()) * 1000)::bigint), true),
      updated_at = now()
  where id = p_app_state_id;
end $$;
```

## Шаг 4 (6.3+) — ответственный автоматически подключается к проекту

Обновляет `assign_work`: когда РП назначает ответственного сотрудника подрядчика на
работу/раздел, этот сотрудник **автоматически добавляется в участники проекта с ролью
«Подрядчик»** (без отдельной ссылки-приглашения) и сразу видит график у себя. Уже
существующего участника (например, admin/member) функция не трогает.

```sql
create or replace function public.assign_work(
  p_app_state_id bigint, p_scope_type text, p_scope_id bigint,
  p_company_id bigint, p_responsible_user_id uuid)
returns bigint language plpgsql security definer as $$
declare v_id bigint; v_project_id bigint; v_email text; v_company_name text;
begin
  select id into v_project_id from public.projects where app_state_id = p_app_state_id;
  if v_project_id is null then raise exception 'Проект не найден'; end if;
  if not public.is_project_admin(v_project_id) then
    raise exception 'Назначать подрядчиков может только руководитель проекта';
  end if;
  if p_scope_type not in ('section','task') then raise exception 'Неверный тип узла'; end if;
  if p_responsible_user_id is not null and not exists(
       select 1 from public.company_members m
       where m.company_id = p_company_id and m.user_id = p_responsible_user_id) then
    raise exception 'Ответственный должен быть сотрудником выбранной компании';
  end if;

  delete from public.work_assignments
    where app_state_id = p_app_state_id and scope_type = p_scope_type and scope_id = p_scope_id;
  insert into public.work_assignments(app_state_id, scope_type, scope_id, company_id, responsible_user_id, assigned_by)
  values (p_app_state_id, p_scope_type, p_scope_id, p_company_id, p_responsible_user_id, auth.uid())
  returning id into v_id;

  -- ответственный автоматически становится участником проекта (подрядчиком), чтобы
  -- видеть график; уже существующего участника с другой ролью не понижаем
  if p_responsible_user_id is not null then
    select email into v_email from auth.users where id = p_responsible_user_id;
    select name into v_company_name from public.companies where id = p_company_id;
    insert into public.project_members(project_id, user_id, role, contractor_name, user_email)
    values (v_project_id, p_responsible_user_id, 'contractor', v_company_name, v_email)
    on conflict (project_id, user_id) do nothing;
  end if;

  return v_id;
end $$;
```

## Шаг 5 (6.5) — автоочистка подрядчика при снятии/смене назначения

Когда РП снимает подрядчика с работ или меняет ответственного, прежний ответственный
убирается из участников проекта — но только если он больше **не отвечает ни за одну
работу** этого графика (иначе остаётся). Затрагивает лишь роль `contractor` (admin/member
и вручную приглашённых не трогает — их никогда не «чистим»).

```sql
-- убрать пользователя из участников проекта, если он подрядчик и больше не отвечает
-- ни за одну работу этого графика
create or replace function public.prune_project_contractor(p_project_id bigint, p_user_id uuid)
returns void language plpgsql security definer as $$
declare v_app_state_id bigint;
begin
  if p_user_id is null then return; end if;
  select app_state_id into v_app_state_id from public.projects where id = p_project_id;
  delete from public.project_members pm
  where pm.project_id = p_project_id
    and pm.user_id = p_user_id
    and pm.role = 'contractor'
    and not exists(
      select 1 from public.work_assignments w
      where w.app_state_id = v_app_state_id and w.responsible_user_id = p_user_id
    );
end $$;

-- assign_work: как в Шаге 4, но при смене ответственного прежний ответственный
-- очищается (если больше нигде не назначен)
create or replace function public.assign_work(
  p_app_state_id bigint, p_scope_type text, p_scope_id bigint,
  p_company_id bigint, p_responsible_user_id uuid)
returns bigint language plpgsql security definer as $$
declare v_id bigint; v_project_id bigint; v_email text; v_company_name text; v_old_resp uuid;
begin
  select id into v_project_id from public.projects where app_state_id = p_app_state_id;
  if v_project_id is null then raise exception 'Проект не найден'; end if;
  if not public.is_project_admin(v_project_id) then
    raise exception 'Назначать подрядчиков может только руководитель проекта';
  end if;
  if p_scope_type not in ('section','task') then raise exception 'Неверный тип узла'; end if;
  if p_responsible_user_id is not null and not exists(
       select 1 from public.company_members m
       where m.company_id = p_company_id and m.user_id = p_responsible_user_id) then
    raise exception 'Ответственный должен быть сотрудником выбранной компании';
  end if;

  -- прежний ответственный этого узла — кандидат на очистку, если сменится
  select responsible_user_id into v_old_resp from public.work_assignments
    where app_state_id = p_app_state_id and scope_type = p_scope_type and scope_id = p_scope_id
    limit 1;

  delete from public.work_assignments
    where app_state_id = p_app_state_id and scope_type = p_scope_type and scope_id = p_scope_id;
  insert into public.work_assignments(app_state_id, scope_type, scope_id, company_id, responsible_user_id, assigned_by)
  values (p_app_state_id, p_scope_type, p_scope_id, p_company_id, p_responsible_user_id, auth.uid())
  returning id into v_id;

  if p_responsible_user_id is not null then
    select email into v_email from auth.users where id = p_responsible_user_id;
    select name into v_company_name from public.companies where id = p_company_id;
    insert into public.project_members(project_id, user_id, role, contractor_name, user_email)
    values (v_project_id, p_responsible_user_id, 'contractor', v_company_name, v_email)
    on conflict (project_id, user_id) do nothing;
  end if;

  if v_old_resp is not null and v_old_resp is distinct from p_responsible_user_id then
    perform public.prune_project_contractor(v_project_id, v_old_resp);
  end if;
  return v_id;
end $$;

-- удаление назначений (одно или каскад раздела) с очисткой освободившихся подрядчиков
create or replace function public.remove_work_assignments(p_app_state_id bigint, p_ids bigint[])
returns void language plpgsql security definer as $$
declare v_project_id bigint; v_resps uuid[];
begin
  select id into v_project_id from public.projects where app_state_id = p_app_state_id;
  if v_project_id is null then raise exception 'Проект не найден'; end if;
  if not public.is_project_admin(v_project_id) then
    raise exception 'Убирать назначения может только руководитель проекта';
  end if;

  select array_agg(distinct responsible_user_id) into v_resps
  from public.work_assignments
  where id = any(p_ids) and app_state_id = p_app_state_id and responsible_user_id is not null;

  delete from public.work_assignments where id = any(p_ids) and app_state_id = p_app_state_id;

  if v_resps is not null then
    perform public.prune_project_contractor(v_project_id, u) from unnest(v_resps) as u;
  end if;
end $$;
```

На этом Этап 6 завершён. Отдельно осталась только миграция старого текстового поля
`t.contractor` в назначения (по конкретным ИНН) — она ручная (нужно знать ИНН каждой
компании), делается по мере необходимости.
